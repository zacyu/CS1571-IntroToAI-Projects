Parent(x,y) ^ Parent(x,z) ^ Distinct(y,z) -> Sibling(y,z)
Parent(x,y) ^ Sibling(x,z) ^ Parent(z,w) -> Cousin(y,w)
Distinct(x,y) -> Distinct(y,x)
Parent(Lisa,Eric)
Parent(Lisa,Rachel)
Parent(Speed,Lisa)
Parent(Speed,Jay)
Parent(Jay,Frances)
Distinct(Eric,Rachel)
Distinct(Lisa,Jay)
PROVE Sibling(Lisa,Jay)
